{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Taxonomy index and counts map: - Return per-taxonomy dictionaries of term counts and sorted index with URL-safe keys. */ -}} {{- $result := dict -}} {{- range $taxonomy, $terms := site.Taxonomies -}} {{- /* Map of counts per taxonomy term (e.g. events: 123). */ -}} {{- $map := dict -}} {{- $keys := slice -}} {{- range $name, $term := $terms -}} {{- $key := $name | urlize -}} {{- $keys = $keys | append $key -}} {{- $map = merge $map (dict $key ($terms.Count $name)) -}} {{- end -}} {{- /* Map of terms index (e.g. events: 0, features: 1). */ -}} {{- $keys = sort $keys -}} {{- $terms_index := dict -}} {{- range $i, $k := $keys -}} {{- $terms_index = merge $terms_index (dict $k $i) -}} {{- end -}} {{- $result = merge $result (dict $taxonomy $map (printf "%s_index" $taxonomy) $terms_index ) -}} {{- end -}} {{- return $result -}}
