{{- /* SPDX-License-Identifier: MIT SPDX-FileCopyrightText: 2026 FreeCAD SPDX-FileNotice: Part of the Trigo theme for Hugo. */ -}} {{- /* Image resource processing: - Support local or remote resources. - Responsive resizing for processable images. @returns {template.HTML} */ -}} {{- $img := .image -}} {{- $src := $img.RelPermalink -}} {{- $alt := .alt | default "" -}} {{- $attributes := .attributes | default dict -}} {{- $format := site.Params.resize_format | default "webp" -}} {{- $resize_dimensions := site.Params.resize_dimensions.cover | default (slice "500x250" "1000x500") -}} {{- $width := "" -}} {{- $height := "" -}} {{- $srcset := slice -}} {{- $sizes := slice -}} {{- /* Responsive and processable image (e.g. bmp, gif, jpg, png, tif, webp). */ -}} {{- if reflect.IsImageResourceProcessable $img -}} {{- $width = $img.Width -}} {{- $height = $img.Height -}} {{- range $resize_dimensions -}} {{- $w := int (index (split . "x") 0) -}} {{- if lt $w $width -}} {{- $resized := $img.Resize (printf "%dx %s" $w $format) -}} {{- $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink $w) -}} {{- $sizes = $sizes | append (printf "(max-width: %dpx) %dpx" $w $w) -}} {{- end -}} {{- end -}} {{- $srcset = $srcset | append (printf "%s %dw" $src $width) -}} {{- $sizes = $sizes | append "75vw" -}} {{- /* Non-processable and non-resizable image with metadata (e.g. avif, heic, heif). */ -}} {{- else if reflect.IsImageResourceWithMeta $img -}} {{- $width = $img.Width -}} {{- $height = $img.Height -}} {{- end -}}
