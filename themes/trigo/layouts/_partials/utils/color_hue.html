"taxonomy_counts" -}} {{- $categories := .Params.categories | default "" -}} {{- $category := cond (reflect.IsSlice $categories) (index $categories 0) $categories -}} {{- $category = replace (string $category) " " "-" -}} {{- $index := index $data.categories_index $category | default 0 -}} {{- $hue = mul 2 (add $index 1) $step -}} {{- else -}} {{- range $index, $page := $section.Pages -}} {{- if eq $.RelPermalink $page.RelPermalink -}} {{- $hue = mul (add $index 1) $step -}} {{- break -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $hue -}}
